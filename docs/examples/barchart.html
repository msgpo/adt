prepare data and size

// Generate some data
var data = [
    {x: 'apple', y: 13},
    {x: 'banana', y: 4},
    {x: 'mango', y: 6},
    {x: 'orange', y: 2},
    {x: 'melon', y: 11},
    {x: 'kiwi', y: 7},
    {x: 'pear', y: 3}
];
var colors = {
    apple: "lime",
    banana: "gold",
    mango: "tomato",
    orange: "orange",
    melon: "green",
    kiwi: "limegreen",
    pear: "khaki"
};

// Read chart size from #widget div
var width = parseFloat(d3.select("#widget").style("width"));
var height = parseFloat(d3.select("#widget").style("height"));

create and render chart

// Add chart to the div with id='widget'
// Use chaining to set parameters
var chart = new adt.widgets.BarChart("mychart", d3.select("#widget"))
    .data(data)                                         // Bind data
    .width(width)                                       // Dimensions
    .height(height)
    .xLabel("fruit name")                               // Labels
    .yLabel("stock")
    .margins(40)                                        // All margins are set to 40px
    .fontSize(14)                                       // Set font size to 14 px
    .colors(colors)                                     // Set colors as key-value pairs
    .render();                                          // Finally, render chart
