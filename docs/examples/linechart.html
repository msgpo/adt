prepare data and size

// Generate some data with errors
var data = Array.from(new Array(100).keys()).map(function(i) {
    return {
        x: i,
        y: {
            first: 10 * Math.pow(i/40, 2),
            second: 10 * Math.log(i+1) * Math.pow(Math.cos(Math.PI*i/15), 2) + 2
        },
        dy: {
            first: i/10,
            second: 6
        }
    }
});

// Read chart size from #widget div
var width = parseFloat(d3.select("#widget").style("width"));
var height = parseFloat(d3.select("#widget").style("height"));

create and render chart

// Add chart to the div with id='widget'
// Use chaining to set parameters
var chart = new adt.widgets.AreaChart("mychart", d3.select("#widget"))
    .data(data)                                         // Bind data
    .width(width)                                       // Dimensions
    .height(height)
    .xLabel("day")                                      // Labels
    .yLabel("price in SEK")
    .margins(40)                                        // All margins are set to 40px
    .fontSize(14)                                       // Set font size to 14 px
    .colors({first: "dodgerblue", second: "limegreen"}) // Set colors as key-value pairs
    .render();                                          // Finally, render chart

interactions

// Add interactions
chart
    .mouseover(function(k) {
        // Highlight hovered plot
        chart.highlight(k, 300);
    })
    .mouseleave(function() {
        // Remove highlights by passing null
        chart.highlight(null, 300);
    })
    .click(function(k) {
        // Show an alert on click
        alert("Plot clicked: '" + k + "'");
    })
    .render();

additional features

// Add trend marker
var marker = chart.addMarker("mymarker", "first", 50, 89, "trend marker");
