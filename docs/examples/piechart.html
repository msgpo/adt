prepare data and size

// Generate some data
var data = [
    {name: "orange", value: 1, color: "orange"},
    {name: "mango", value: 4, color: "tomato"},
    {name: "banana", value: 4, color: "gold"},
    {name: "apple", value: 2, color: "limegreen"}
];
var colors = {
    lime: "lime",
    banana: "gold",
    orange: "orange",
    apple: "limegreen"
};

// Read chart size from #widget div
var width = parseFloat(d3.select("#widget").style("width"));
var height = parseFloat(d3.select("#widget").style("height"));

create and render chart

// Add chart
var chart = new adt.widgets.PieChart("mychart", d3.select("#widget"))
    .data(data)                         // Bind data
    .x(width/2 - height/2 + 20)         // Dimensions
    .innerRadius(30)
    .outerRadius(height / 2 - 20)
    .xLabel("fruits")                   // Add label
    .margins(30)                        // Margins are set to 30px
    .fontSize(14)                       // Font is set to 14px
    .colors(colors)
    .render();                          // Render chart

interactions

// Add interactions
chart
    .mouseover(function(k) {
        // Highlight hovered bar
        chart.highlight(k, 300);
    })
    .mouseleave(function() {
        // Remove highlights by passing null
        chart.highlight(null, 300);
    })
    .click(function(k) {
        // Show an alert on click
        alert("Plot clicked: '" + k + "'");
    })
    .render();
